### Importing and cleaning the unemployment Data.

``` r
unemploy_data <-
  unemploy_data <-
  read_csv("./fivethirtyeight_datasets/unemployment.csv") %>%
  janitor::clean_names() %>%
  pivot_longer(
    jan:dec,
    names_to = "month",
    values_to = "percentage"

  mutate(across(.col = c(year), as.integer)) %>%
```
